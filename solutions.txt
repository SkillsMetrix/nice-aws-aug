package com.app;

import java.util.Collection;
import java.util.Collections;

import com.amazonaws.services.lambda.runtime.Context;
import com.amazonaws.services.lambda.runtime.RequestHandler;

public class MainApp implements RequestHandler<Object, GatewayResponse>{

	@Override
	public GatewayResponse handleRequest(Object input, Context context) {
		String message= "Welcome to Nice Pune";
		System.out.println(message);
		GatewayResponse response= new GatewayResponse(message, 200, Collections.singletonMap
				("POWERED-BY", "Tech Platforms"));
		return response;
	}
	
	

}
